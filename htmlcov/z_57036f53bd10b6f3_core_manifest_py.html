161def _expand_signatures( 

162 *, domain_ranges: tuple[PredicateDomainRange, ...] 

163) -> tuple[PredicateSignatureDefinition, ...]: 

164 """Expand the compact domain/range unions into concrete signature rows. 

165 

166 Product rows pair every subject type with every object type in subject-major 

167 order; same-kind rows pair each core type with itself in display order. The 

168 expansion is deterministic and yields exactly the 116 core-v1 rows, asserted 

169 by `_assert_manifest_integrity`. 

170 """ 

171 signatures: list[PredicateSignatureDefinition] = [] 

172 for domain_range in domain_ranges: 

173 pairs = ( 

174 tuple((name, name) for name in domain_range.subject_types) 

175 if domain_range.same_kind 

176 else tuple( 

177 (subject, obj) 

178 for subject in domain_range.subject_types 

179 for obj in domain_range.object_types 

180 ) 

181 ) 

182 signatures.extend( 

183 PredicateSignatureDefinition( 

184 predicate=domain_range.predicate, subject_type=subject, object_type=obj 

185 ) 

186 for subject, obj in pairs 

187 ) 

188 return tuple(signatures)

575def _assert_manifest_integrity(manifest: CoreManifest) -> None: 

576 """Fail import if the packaged manifest loses a binding core-v1 invariant.""" 

577 entity_keys = tuple(entity.type for entity in manifest.entity_types) 

578 predicate_keys = tuple(predicate.predicate for predicate in manifest.predicates) 

579 signature_keys = tuple( 

580 (signature.predicate, signature.subject_type, signature.object_type) 

581 for signature in manifest.predicate_signatures 

582 ) 

583 

584 assert manifest.manifest_version == "core-v1" 

585 assert len(entity_keys) == len(set(entity_keys)) == 8 

586 assert len(predicate_keys) == len(set(predicate_keys)) == 16 

587 assert len(signature_keys) == len(set(signature_keys)) == 116 

588 assert all(entity.parent_type is None for entity in manifest.entity_types) 

589 assert predicate_keys[0] == "related_to" 

590 assert set( 

591 signature.predicate for signature in manifest.predicate_signatures 

592 ) <= set(predicate_keys) 

593 assert {signature.subject_type for signature in manifest.predicate_signatures} | { 

594 signature.object_type for signature in manifest.predicate_signatures 

595 } <= set(entity_keys)
