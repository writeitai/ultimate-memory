391class EnvelopePart(BaseModel): 

392 """One single-grain section of a composite answer (S47). 

393 

394 A mixed answer — S47's "everything Alice *said* about pricing, plus what 

395 we *believe*" — is EXPLICITLY two-part, never blended: each part carries 

396 its own grain and its own single-grain results, so the fact/evidence 

397 discipline is never diluted. Single-grain answers skip `parts` and read 

398 flat off the top-level fields. 

399 """ 

400 

401 model_config = ConfigDict(frozen=True, extra="forbid") 

402 

403 grain: Grain # strictly single-grain: fact | evidence | compiled (S47) 

404 label: str | None = None # e.g. "said" vs "believed" 

405 facts: tuple[FactResult, ...] = () 

406 evidence: tuple[EvidenceResult, ...] = () 

407 sources: tuple[SourceRecord, ...] = () 

408 nodes: tuple[GraphNode, ...] = () 

409 aggregate: AggregateReport | None = None 

410 pages: tuple[PageRef, ...] = () 

411 truncation: Truncation | None = None 

412 

413 @model_validator(mode="after") 

414 def _payload_matches_grain(self) -> "EnvelopePart": 

415 """A part is strictly single-grain: it carries only the payload its 

416 own grain owns — a fact part holds facts (and graph nodes / an 

417 aggregate), an evidence part holds claims, a compiled part holds K 

418 pages. Carrying another grain's payload, or being composite, is the 

419 blending S47 forbids. `sources` (hydration handles) is cross-grain.""" 

420 if self.grain is Grain.COMPOSITE: 

421 raise ValueError("a composite part is not single-grain (S47)") 

422 owned = { 

423 Grain.FACT: {"facts", "nodes", "aggregate"}, 

424 Grain.EVIDENCE: {"evidence"}, 

425 Grain.COMPILED: {"pages"}, 

426 }[self.grain] 

427 populated = { 

428 name 

429 for name, value in ( 

430 ("facts", self.facts), 

431 ("evidence", self.evidence), 

432 ("nodes", self.nodes), 

433 ("aggregate", self.aggregate), 

434 ("pages", self.pages), 

435 ) 

436 if value 

437 } 

438 stray = populated - owned 

439 if stray: 

440 raise ValueError( 

441 f"a {self.grain.value}-grain part carries {sorted(stray)}" 

442 " belonging to another grain (S47)" 

443 ) 

444 return self

447class Envelope(BaseModel): 

448 """The D49 envelope: results plus the answer's machine-readable self-account. 

449 

450 A single-grain answer (the common case) reads flat: the top-level 

451 `grain` and the matching result tuple. A `composite` answer sets 

452 `grain = composite` and carries `parts[]`, each strictly single-grain 

453 (S47) — so a caller never has to disentangle blended grains. 

454 """ 

455 

456 model_config = ConfigDict(frozen=True, extra="forbid") 

457 

458 grain: Grain 

459 parts: tuple["EnvelopePart", ...] = () # S47: composite ⇒ read parts[] 

460 as_of_valid_at: UTCDateTime | None = None # echo of the applied valid_at 

461 as_of_believed_at: UTCDateTime | None = None # echo of the applied believed_at 

462 identity_regime: IdentityRegime = IdentityRegime.CURRENT # S61: which regime 

463 entities: tuple[EntityCandidate, ...] = () 

464 facts: tuple[FactResult, ...] = () 

465 evidence: tuple[EvidenceResult, ...] = () 

466 sources: tuple[SourceRecord, ...] = () 

467 transcript: tuple["TranscriptEntry", ...] = () # S8: the audit surface 

468 nodes: tuple[GraphNode, ...] = () # S18: neighborhood members 

469 paths: tuple[GraphPath, ...] = () # S17/S21: compound connections 

470 edges: tuple[GraphEdge, ...] = () # the traversed relations 

471 ranking: tuple[RankedItem, ...] = () # S46: fused / reranked order 

472 changes: tuple[ChangeRecord, ...] = () # S13/S14/S30: the delta feed 

473 aggregate: AggregateReport | None = None # S26–S30/S40: enumerated only 

474 pages: tuple[PageRef, ...] = () # S31/S45: pages_about discovery 

475 freshness: Freshness 

476 truncation: Truncation | None = None # S18/S49: caps are never silent 

477 dropped_by_hydration: int = 0 

478 negative: Negative | None = None

480 @model_validator(mode="after") 

481 def _composite_uses_parts(self) -> "Envelope": 

482 """Enforce the S47 discipline: `parts` belong only to a composite 

483 answer, and a composite's data lives IN its parts, never blended into 

484 the top-level result tuples. (Each part's own single-grain purity is 

485 checked on `EnvelopePart`.) A flat compound answer with no parts — 

486 hydrate's fact-with-evidence bundle — is untouched.""" 

487 if not self.parts: 

488 return self 

489 if self.grain is not Grain.COMPOSITE: 

490 raise ValueError("only a composite envelope carries parts[]") 

491 blended = self.aggregate is not None or any( 

492 ( 

493 self.entities, 

494 self.facts, 

495 self.evidence, 

496 self.sources, 

497 self.transcript, 

498 self.nodes, 

499 self.paths, 

500 self.edges, 

501 self.ranking, 

502 self.changes, 

503 self.pages, 

504 ) 

505 ) 

506 if blended: 

507 raise ValueError( 

508 "a composite answer's data lives in parts[], never blended" 

509 " into the top-level result tuples (S47)" 

510 ) 

511 return self
