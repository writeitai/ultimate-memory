723_BUDGET_WINDOW_SPEND = text( 

724 """ 

725 WITH bounds AS ( 

726 SELECT 

727 to_timestamp( 

728 floor(extract(epoch FROM now()) / :window_seconds) 

729 * :window_seconds 

730 ) AS window_started_at, 

731 to_timestamp( 

732 (floor(extract(epoch FROM now()) / :window_seconds) + 1) 

733 * :window_seconds 

734 ) AS window_ends_at 

735 ) 

736 SELECT bounds.window_started_at, 

737 bounds.window_ends_at, 

738 COALESCE(sum(cost_ledger.cost_usd), 0) AS spent_usd 

739 FROM bounds 

740 LEFT JOIN cost_ledger 

741 ON cost_ledger.deployment_id = :deployment_id 

742 AND cost_ledger.stage = :stage 

743 AND cost_ledger.lane IS NOT DISTINCT FROM :lane 

744 AND cost_ledger.occurred_at >= bounds.window_started_at 

745 AND cost_ledger.occurred_at < bounds.window_ends_at 

746 GROUP BY bounds.window_started_at, bounds.window_ends_at 

747 """
