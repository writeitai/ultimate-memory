597 """ 

598 SELECT cl.claim_id, c.version_id AS from_version_id 

599 FROM claims cl 

600 JOIN chunks c ON c.chunk_id = cl.chunk_id 

601 WHERE cl.deployment_id = :deployment_id 

602 AND cl.doc_id = :doc_id 

603 AND cl.is_current_testimony 

604 AND c.version_id <> :current_version_id 

605 AND NOT EXISTS (SELECT 1 FROM chunk_claims cc 

606 JOIN chunks cur ON cur.chunk_id = cc.chunk_id 

607 WHERE cc.claim_id = cl.claim_id
